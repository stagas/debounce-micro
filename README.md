## Install

```sh
$ npm i debounce-micro
```

## Example

```ts
let x = 0
const fn = debounce((y: number) => (x += y))
fn(1)
fn(2)
expect(x).toEqual(0)
queueMicrotask(() => {
  expect(x).toEqual(2) // instead of 3
  done()
})
```
